## This will give conditions under policy id
POLICY_ID="5631237"

newrelic nerdgraph query '{
  actor {
    account(id: 2547913) {
      alerts {
        nrqlConditionsSearch {
          nrqlConditions {
            id
            name
            policyId
          }
        }
      }
    }
  }
}' | jq -r --arg pid "$POLICY_ID" '
  .actor.account.alerts.nrqlConditionsSearch.nrqlConditions[]
  | select(.policyId == $pid)
  | [.id, .name] | @tsv'
